// Handle change in description
const handleDescriptionChange = (skillName: string, description: string) => {
  setSkillsList(prev => 
    prev.map(skill => 
      skill.name === skillName 
        ? { ...skill, description } 
        : skill
    )
  );
};

// Add the Description TableHead in each TableHeader section (technical and functional tabs)
// Add this after the Level TableHead in both places:
<TableHead>Description</TableHead>

// Add this Textarea input field in the skill rows
// Insert this after the Level dropdown in the skill row sections
<TableCell>
  <Textarea
    placeholder="Describe your experience with this skill..."
    value={skillsList.find(s => s.name === skill.name)?.description || ""}
    onChange={(e) => handleDescriptionChange(skill.name, e.target.value)}
    disabled={isDisabled || !selectedSkills[skill.name]}
    className="w-full max-w-xs h-24"
  />
</TableCell>
